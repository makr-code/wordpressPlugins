SELECT u.name, e.score, g.friends
FROM users u
JOIN embeddings e ON e.user_id = u.id

-- ThemisDB SQL: Vector + Relational Query
SELECT p.title, p.author, p.published_at,
       embedding_distance(p.vector, query.vec) AS score
FROM   posts p,
       (
         SELECT embed('AI database performance') AS vec
       ) query
WHERE  p.published_at > NOW() - INTERVAL '30 days'
ORDER BY score ASC
LIMIT  5;
POST /api/v3/query
Content-Type: application/json

{
  "collection": "posts",
  "vector_search": {
    "query": "AI database performance",
    "limit": 5,
    "threshold": 0.8
  },
  "filter": {
    "published_at": { "$gte": "2025-01-01" }
  }
}
import themisdb

db = themisdb.connect("themisdb://localhost:5432/mydb")

results = db.posts.vector_search(
    query="AI database performance",
    limit=5,
    filter={"published_at": {"$gte": "2025-01-01"}}
)

for post in results:
    print(f"{post.title} — score: {post.score:.3f}")
